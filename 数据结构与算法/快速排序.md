function quickSort(arr) {
  if (arr.length < 2) return arr;

  let left = [];
  let right = [];
  let target = arr[0];

  for (let i = 1; i < arr.length; i++) {
    arr[i] > target ? right.push(arr[i]) : left.push(arr[i]);
  }

  return [...quickSort(left), target, ...quickSort(right)];
}
